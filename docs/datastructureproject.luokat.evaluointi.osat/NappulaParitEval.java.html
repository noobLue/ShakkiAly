package datastructureproject.luokat.evaluointi.osat;

import chess.model.Side;
import datastructureproject.luokat.nappulat.Lahetti;
import datastructureproject.luokat.nappulat.Nappula;
import datastructureproject.luokat.nappulat.Ratsu;
import datastructureproject.luokat.nappulat.Torni;

public class NappulaParitEval implements OsaEvaluaattori {
    private static final double ARVOKERROIN = 0.1d; 
    private Side puoli;

    private int oTornit;
    private int oRatsut;
    private int oLahetit;
    private int vTornit;
    private int vRatsut;
    private int vLahetit;

    public NappulaParitEval(Side puoli) {
        this.puoli = puoli;

        this.oTornit = 0;
        this.oRatsut = 0;
        this.oLahetit = 0;

        this.vTornit = 0;
        this.vRatsut = 0;
        this.vLahetit = 0;
    }

    private void oProsessointi(Nappula nappula) {
        if (nappula instanceof Torni) {
            oTornit++;
        } else if (nappula instanceof Ratsu) {
            oRatsut++;
        } else if (nappula instanceof Lahetti) {
            oLahetit++;
        }
    }

    private void vProsessointi(Nappula nappula) {
        if (nappula instanceof Torni) {
            vTornit++;
        } else if (nappula instanceof Ratsu) {
            vRatsut++;
        } else if (nappula instanceof Lahetti) {
            vLahetit++;
        }
    }

    @Override
    public void prosessoiNappula(Nappula nappula) {
        if (nappula.getPuoli() == puoli) {
            oProsessointi(nappula);
        } else {
            vProsessointi(nappula);
        }
    }

    @Override
    public double getArvo() {

        int laskuri = 0;
        if (oTornit == 2) {
            laskuri--;
        }
        if (oRatsut == 2) {
            laskuri--;
        } 
        if (oLahetit == 2) {
            laskuri++;
        }

        if (vTornit == 2) {
            laskuri++;
        }
        if (vRatsut == 2) {
            laskuri++;
        } 
        if (vLahetit == 2) {
            laskuri--;
        }

        return (double)laskuri * ARVOKERROIN ;
    }
    
}
